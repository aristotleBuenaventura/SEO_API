        if(cache[m])return;
        try{
            const r=await fetch("' . admin_url('admin-ajax.php?action=fetch_helpslot_games') . '&manuf="+m);
            const d=await r.json();
            cache[m]=d;
            const c=document.getElementById("helpslot-"+m);
            if(c)c.innerHTML=renderGamesHTML(d,m);
            if(isSearchMode)applySearchFilter();
        }catch(e){
            console.error(e);
            const c=document.getElementById("helpslot-"+m);
            if(c)c.innerHTML="<p style=\\"color:#f44;text-align:center;padding:40px;\\">Error loading "+m+"</p>";
        }
    }

        const hasMore=d.length>INITIAL_GAME_LIMIT;

            let progressPercent=g.value!=null?g.value.toFixed(2):0;

            "<div class=\\"game-card-left\\">"+

    function toggleSeeMore(manuf){
        expandedManufs[manuf]=true;
        const container=document.getElementById("helpslot-"+manuf);
        if(!container)return;
        
        const hiddenCards=container.querySelectorAll(".game-card.hidden");
        hiddenCards.forEach(function(card){
            card.classList.remove("hidden");
        });
        
        const btn=container.querySelector(".see-more-btn");
        if(btn)btn.classList.add("hidden");
    }

    function handleImageLoad(img){
        img.classList.remove("loading");
        img.classList.add("loaded");
        const loader=img.parentElement.querySelector(".loader");
        if(loader)loader.classList.add("hidden");
    }

    function handleImageError(img){
        img.classList.add("error");
        const loader=img.parentElement.querySelector(".loader");
        if(loader)loader.classList.add("hidden");
        const wrapper=img.parentElement;
        if(!wrapper.querySelector(".placeholder-text")){
            const placeholder=document.createElement("div");
            placeholder.className="placeholder-text";
            placeholder.style.cssText="font-size:10px;text-align:center;color:#999;";
            placeholder.textContent="No Image";
            wrapper.appendChild(placeholder);
        }
    }

    async function showManuf(m){
        isSearchMode=false;
        document.getElementById("helpslot-search").value="";
        searchTerm="";
        document.getElementById("helpslot-search-results").style.display="none";
        document.getElementById("helpslot-container").style.display="block";
        currentManuf=m;
        document.querySelectorAll("#helpslot-buttons button").forEach(function(b){
            b.classList.toggle("active",b.getAttribute("data-manuf")===m);
        });
        manufacturers.forEach(function(x){
            let el=document.getElementById("helpslot-"+x);
            if(el)el.style.display="none";
        });
        let selectedContainer=document.getElementById("helpslot-"+m);
        if(selectedContainer){
            selectedContainer.style.display="block";
        }

            await loadManuf(m);
        }
        stopRefreshLoop();
        startRefreshLoop();
    }

    function applySearchFilter(){
        const searchResults=document.getElementById("helpslot-search-results");
        const mainContainer=document.getElementById("helpslot-container");
        
        if(searchTerm===""){
            isSearchMode=false;
            searchResults.style.display="none";
            mainContainer.style.display="block";
            manufacturers.forEach(function(m){
                let el=document.getElementById("helpslot-"+m);
                if(el)el.style.display=(m===currentManuf?"block":"none");
            });
            document.querySelectorAll(".no-results").forEach(el=>el.remove());
            return;
        }
        
        isSearchMode=true;
        mainContainer.style.display="none";
        searchResults.style.display="block";
        
        const fragment=document.createDocumentFragment();
        const gridDiv=document.createElement("div");
        gridDiv.className="game-grid";
        
        let matchCount=0;
        
        manufacturers.forEach(function(m){
            let container=document.getElementById("helpslot-"+m);
            if(!container)return;
            
            const cards=container.querySelectorAll(".game-card");
            cards.forEach(function(card){
                const nameNoSpace=card.getAttribute("data-name-nospace");
                if(nameNoSpace.includes(searchTerm)){
                    const clonedCard=card.cloneNode(true);
                    clonedCard.classList.remove("hidden");
                    gridDiv.appendChild(clonedCard);
                    matchCount++;
                }
            });
        });
        
        if(matchCount>0){
            fragment.appendChild(gridDiv);
            searchResults.innerHTML="";
            searchResults.appendChild(fragment);
        }else{
            searchResults.innerHTML="<div class=\\"no-results\\">No games found matching \\""+document.getElementById("helpslot-search").value+"\\"</div>";
        }
        
        document.querySelectorAll("#helpslot-buttons button").forEach(function(b){
            b.classList.remove("active");
        });
    }

    function startRefreshLoop(){
        if(refreshTimer)clearTimeout(refreshTimer);
        refreshLoop();
    }

    function stopRefreshLoop(){
        if(refreshTimer){
            clearTimeout(refreshTimer);
            refreshTimer=null;
        }
    }

    async function refreshLoop(){
        if(isRefreshing){
            refreshTimer=setTimeout(refreshLoop,1000);
            return;
        }
        isRefreshing=true;
        try{
            const manufToRefresh=isSearchMode?manufacturers:[currentManuf];
            for(let m of manufToRefresh){
                if(!cache[m])continue;
                const r=await fetch("' . admin_url('admin-ajax.php?action=fetch_helpslot_values') . '&manuf="+m);
                const d=await r.json();
                if(!Array.isArray(d))continue;
                d.forEach(function(g){
                    let c=document.querySelector("#helpslot-"+m+" .game-card[data-id=\'"+g.id+"\']");
                    updateGameCard(c,g);
                    if(isSearchMode){
                        let searchCard=document.querySelector("#helpslot-search-results .game-card[data-id=\'"+g.id+"\'][data-manuf=\'"+m+"\']");
                        updateGameCard(searchCard,g);
                    }
                });
            }
        }catch(e){
            console.error(e);
        }finally{
            isRefreshing=false;
            refreshTimer=setTimeout(refreshLoop,isSearchMode?5000:1000);
        }
    }

    function updateGameCard(c,g){
        if(!c)return;
        c.querySelector(".val-value").innerText=g.value.toFixed(2)+"%";
        c.querySelector(".val-min10").innerHTML=formatPercent(g.min10);
        c.querySelector(".val-hr1").innerHTML=formatPercent(g.hr1);
        c.querySelector(".val-hr3").innerHTML=formatPercent(g.hr3);
        c.querySelector(".val-hr6").innerHTML=formatPercent(g.hr6);
        let progressBar=c.querySelector(".game-card-left > div > div:last-child");
        let progressText=c.querySelector(".game-card-left > div > div:first-child");
        if(progressBar){
            progressBar.style.width=g.value.toFixed(2)+"%";
            progressBar.style.backgroundColor=g.up?"green":"red";
        }
        if(progressText){
            progressText.innerText=g.value.toFixed(2)+"%";
        }
    }

    function formatPercent(v){
        if(v==null)return"";
        return"<span style=\'color:"+(v>=0?"green":"red")+"\'>"+v.toFixed(2)+"%</span>";
    }

    document.addEventListener("DOMContentLoaded",function(){
        new Swiper(".helpslot-swiper",{slidesPerView:"auto",spaceBetween:10,freeMode:false,centerInsufficientSlides:true});
        
        const searchInput=document.getElementById("helpslot-search");
        searchInput.addEventListener("input",function(e){
            if(searchDebounceTimer){
                clearTimeout(searchDebounceTimer);
            }
            
            const rawValue=e.target.value.toLowerCase().trim();
            
            searchDebounceTimer=setTimeout(function(){
                searchTerm=rawValue.replace(/\\s+/g,"");
                applySearchFilter();
            },DEBOUNCE_DELAY);
        });
        
        loadAllManufs();
    });
    </script>
    ';
    return $html;
}

add_action("wp_ajax_fetch_helpslot_games","fetch_helpslot_games");
add_action("wp_ajax_nopriv_fetch_helpslot_games","fetch_helpslot_games");
function fetch_helpslot_games(){
    $manuf=isset($_GET["manuf"])?sanitize_text_field($_GET["manuf"]):"JILI";
    $all_games=[];
    $current=1;
    do{
        $res=wp_remote_get("https://helpslot.win/api/games/?manuf=".$manuf."&current=".$current);
        if(is_wp_error($res))break;
        $data=json_decode(wp_remote_retrieve_body($res),true);
        $games=$data["data"]??[];
        if(empty($games))break;
        $all_games=array_merge($all_games,$games);
        $current++;
    }while(!empty($games));
    wp_send_json($all_games);
}

add_action("wp_ajax_nopriv_fetch_helpslot_values","fetch_helpslot_values");
function fetch_helpslot_values(){
    $manuf=isset($_GET["manuf"])?sanitize_text_field($_GET["manuf"]):"JILI";
    $all_games=[];
    $current=1;
    do{
        $res=wp_remote_get("https://helpslot.win/api/games/?manuf=".$manuf."&current=".$current);
        if(is_wp_error($res))break;
        $data=json_decode(wp_remote_retrieve_body($res),true);
        $games=$data["data"]??[];
        if(empty($games))break;
        $all_games=array_merge($all_games,$games);
        $current++;
    }while(!empty($games));
    wp_send_json($all_games);
}
