    function dum_sheet_effective_source_url() {
        $u = defined('DUM_SHEET_SOURCE_URL') ? (string) DUM_SHEET_SOURCE_URL : '';
        /**
         * Filter the Google Sheet / CSV source URL.
         *
         * @param string $u Raw URL from DUM_SHEET_SOURCE_URL.
         */
        return apply_filters('dum_sheet_source_url', trim($u));
    }

    function dum_normalize_sheet_csv_url($url) {
        $url = trim((string) $url);
        if ($url === '') {
            return '';
        }
        // Already a publish or export CSV-style URL.
        if (preg_match('#/spreadsheets/d/e/#', $url)) {
            return $url;
        }
        if (preg_match('~/export\?[^#]*format=csv~i', $url) || preg_match('~[?&]output=csv(\b|&)~i', $url)) {
            return $url;
        }
        // docs.google.com/spreadsheets/d/SHEET_ID/... → public CSV export (needs link sharing: Viewer).
        if (preg_match('#docs\.google\.com/spreadsheets/d/([a-zA-Z0-9_-]+)#', $url, $m)) {
            $gid = '0';
            if (defined('DUM_SHEET_GID') && (string) DUM_SHEET_GID !== '') {
                $gid = (string) DUM_SHEET_GID;
            } elseif (preg_match('~[?#&]gid=(\d+)~', $url, $gm)) {
                $gid = $gm[1];
            }
            return 'https://docs.google.com/spreadsheets/d/' . $m[1] . '/export?format=csv&gid=' . rawurlencode($gid);
        }
        return $url;
    }

    function dum_is_sheet_source_mode() {

    }

    function dum_sheet_csv_fetch_url() {
        return dum_normalize_sheet_csv_url(dum_sheet_effective_source_url());
    }

    /**
     * @return string[] Normalized https://host URLs (unique, in sheet order).
     */
    function dum_fetch_sheet_domain_urls($csv_url) {
        if ($csv_url === '') {
            return [];
        }
        $sslverify = true;
        if (defined('DUM_SHEET_SSL_INSECURE') && DUM_SHEET_SSL_INSECURE) {
            $sslverify = false;
        }
        $req_args = [
            'timeout' => 25,
            'redirection' => 5,
            'sslverify' => $sslverify,
            'headers' => [
                'Accept' => 'text/csv,text/plain,*/*;q=0.8',
            ],
        ];
        /**
         * @param array<string,mixed> $req_args Arguments for wp_remote_get( $csv_url, $req_args ).
         */
        $req_args = apply_filters('dum_sheet_remote_get_args', $req_args, $csv_url);
        $res = wp_remote_get($csv_url, $req_args);
        if (is_wp_error($res)) {
            return [];
        }
        $code = (int) wp_remote_retrieve_response_code($res);
        if ($code < 200 || $code >= 300) {
            return [];
        }
        $body = wp_remote_retrieve_body($res);
        if ($body === null || $body === '') {
            return [];
        }
        $trimmed = ltrim($body);
        // Google sign-in / error HTML instead of CSV.
        if ($trimmed !== '' && ($trimmed[0] === '<' || stripos($trimmed, '<!DOCTYPE') === 0 || stripos($trimmed, '<html') === 0)) {
            return [];
        }
        $lines = preg_split('/\r\n|\r|\n/', $body);
        if (!empty($lines[0])) {
            $lines[0] = preg_replace('/^\xEF\xBB\xBF/', '', $lines[0]);
        }
        $out = [];
        $seen = [];
        $header_labels = ['domain', 'url', 'website', 'site', 'domains', 'link', 'hostname'];
        foreach ($lines as $i => $line) {
            $line = trim((string) $line);
            if ($line === '' || (isset($line[0]) && $line[0] === '#')) {
                continue;
            }
            $parts = str_getcsv($line);
            $max_col = min(5, count($parts));
            $picked = null;
            for ($ci = 0; $ci < $max_col; $ci++) {
                $cell = isset($parts[$ci]) ? trim((string) $parts[$ci]) : '';
                if ($cell === '') {
                    continue;
                }
                $low = strtolower($cell);
                if ($i === 0 && in_array($low, $header_labels, true)) {
                    continue;
                }
                $url = dum_normalize_domain($cell);
                if ($url) {
                    $picked = $url;
                    break;
                }
            }
            if ($picked === null) {
                continue;
            }
            if (isset($seen[$picked])) {
                continue;
            }
            $seen[$picked] = true;
            $out[] = $picked;
        }
        return $out;
    }

    /**
     * Sync DB domains to match the Google Sheet. Does not delete everything if fetch returns empty.
     *
     * @return array{ok:bool, synced:int, fetched:int}
     */
    function dum_sync_domains_from_google_sheet($wpdb, $table_name, $force_fetch = false) {
        $csv_url = dum_sheet_csv_fetch_url();
        if ($csv_url === '') {
            return ['ok' => false, 'synced' => 0, 'fetched' => 0];
        }
        // v2: older versions cached empty arrays and blocked sync for minutes.
        $cache_key = 'dum_sheet_url_list_cache_v2';

        $urls = null;
        if (!$force_fetch) {
            $cached = get_transient($cache_key);
            if (is_array($cached) && isset($cached['urls'], $cached['t']) && (time() - (int) $cached['t']) < 90) {
                $urls = $cached['urls'];
            }
        }
        if ($urls === null) {
            $urls = dum_fetch_sheet_domain_urls($csv_url);
            // Do not cache empty fetches — otherwise one failed request blocks the UI for minutes.
            if (is_array($urls) && !empty($urls)) {
                set_transient($cache_key, ['urls' => $urls, 't' => time()], 120);
            }
        }
        if (!is_array($urls) || empty($urls)) {
            return ['ok' => false, 'synced' => 0, 'fetched' => is_array($urls) ? count($urls) : 0];
        }
        $n = count($urls);
        $placeholders = implode(',', array_fill(0, $n, '%s'));
        $wpdb->query($wpdb->prepare("DELETE FROM $table_name WHERE domain NOT IN ($placeholders)", $urls));
        $existing = $wpdb->get_col("SELECT domain FROM $table_name");
        $have = array_flip($existing ?: []);
        $now = dum_ph_datetime_string();
        $added = 0;
        foreach ($urls as $url) {
            if (isset($have[$url])) {
                continue;
            }
            $wpdb->insert($table_name, [
                'domain' => $url,
                'status' => '0',
                'detail' => 'Pending check',
                'last_checked' => $now,
            ], ['%s', '%s', '%s', '%s']);
            $have[$url] = true;
            $added++;
        }
        return ['ok' => true, 'synced' => $n, 'fetched' => $n];
    }

    function dum_table_name() {

    function dum_ensure_table() {
        global $wpdb;
        $table_name = dum_table_name();
        $wpdb->query("CREATE TABLE IF NOT EXISTS $table_name (
            id INT AUTO_INCREMENT PRIMARY KEY,
            domain VARCHAR(255) UNIQUE,
            status VARCHAR(10),
            detail TEXT,
            last_checked DATETIME
        )");
    }

    function dum_normalize_domain($url) {
        $url = trim($url);
        if (!preg_match('#^https?://#', $url)) {
            $url = 'https://' . $url;
        }
        $parsed = parse_url($url);
        if (!$parsed || empty($parsed['host'])) {
            return false;
        }

        $host = strtolower($parsed['host']);
        $host = preg_replace('/^www\./', '', $host);

        return 'https://' . $host;
    }

    function dum_ph_datetime_string() {
        try {
            $tz = new DateTimeZone('Asia/Manila');
            $now = new DateTime('now', $tz);
            return $now->format('Y-m-d H:i:s');
        } catch (Exception $e) {
            return date('Y-m-d H:i:s');
        }
    }

    function dum_probe_attempt($url, $connect_timeout, $timeout, $prefer_head = true) {
        $ch = curl_init($url);
        curl_setopt_array($ch, [
            CURLOPT_RETURNTRANSFER => true,
            CURLOPT_NOBODY => $prefer_head,
            CURLOPT_FOLLOWLOCATION => true,
            CURLOPT_MAXREDIRS => 5,
            CURLOPT_TIMEOUT => (int) $timeout,
            CURLOPT_CONNECTTIMEOUT => (int) $connect_timeout,
            CURLOPT_SSL_VERIFYPEER => false,
            CURLOPT_SSL_VERIFYHOST => false,
            CURLOPT_USERAGENT => 'SiteUptimeMonitor/1.0 WordPress',
            CURLOPT_IPRESOLVE => CURL_IPRESOLVE_WHATEVER,
        ]);
        curl_exec($ch);
        $errno = curl_errno($ch);
        $err = $errno ? (string) curl_error($ch) : '';
        $code = (int) curl_getinfo($ch, CURLINFO_HTTP_CODE);
        $time = (float) curl_getinfo($ch, CURLINFO_TOTAL_TIME);
        curl_close($ch);

        return [
            'errno' => $errno,
            'error' => $err,
            'http_code' => $code,
            'time' => $time,
        ];
    }

    /**
     * Probe URL: UP if we get a valid HTTP response (any 1xx–5xx from server).
     * DOWN on timeout, connection/DNS/TLS failure, or HTTP code 0.
     */
    function dum_probe_site($url) {
        $attempt = dum_probe_attempt($url, 10, 15, true);

        // Retry once with a longer timeout and full body fetch for borderline servers.
        if (($attempt['http_code'] <= 0 || $attempt['http_code'] >= 600) && (int) $attempt['errno'] !== 0) {
            $retry = dum_probe_attempt($url, 20, 30, false);
            if ((int) $retry['http_code'] > 0 && (int) $retry['http_code'] < 600) {
                $attempt = $retry;
            }
        }

        // Fallback to HTTP only when HTTPS failed entirely.
        if ((int) $attempt['http_code'] === 0 && (int) $attempt['errno'] !== 0) {
            $http_url = preg_replace('#^https://#i', 'http://', $url);
            if ($http_url && $http_url !== $url) {
                $fallback = dum_probe_attempt($http_url, 20, 30, true);
                if ((int) $fallback['http_code'] > 0 && (int) $fallback['http_code'] < 600) {
                    $attempt = $fallback;
                }
            }
        }

        $errno = (int) $attempt['errno'];
        $err = (string) $attempt['error'];
        $code = (int) $attempt['http_code'];
        $time = (float) $attempt['time'];
        $up = ($code > 0 && $code < 600);
        if ($up) {
            $detail = 'HTTP ' . $code;
            if ($time > 0) {
                $detail .= ' · ' . sprintf('%.2fs', $time);
            }
        } else {
            $detail = $err !== '' ? $err : ($code > 0 ? 'HTTP ' . $code : 'No response');
        }

        return [
            'up' => $up,
            'http_code' => $code,
            'detail' => $detail,
        ];
    }

    function dum_upsert_domain($wpdb, $table_name, $url) {
        if (!$url) {
            return null;
        }
        $result = dum_probe_site($url);
        $status = (string) (int) $result['http_code'];
        $detail = (string) $result['detail'];

        $now = dum_ph_datetime_string();

        $existing = $wpdb->get_row($wpdb->prepare(
            "SELECT id FROM $table_name WHERE domain=%s",
            $url
        ));

        if ($existing) {
            $wpdb->update($table_name, [
                'status' => $status,
                'detail' => $detail,
                'last_checked' => $now,
            ], ['id' => $existing->id]);
        } else {
            $wpdb->insert($table_name, [
                'domain' => $url,
                'status' => $status,
                'detail' => $detail,
                'last_checked' => $now,
            ]);
        }

        return $wpdb->get_row($wpdb->prepare(
            "SELECT * FROM $table_name WHERE domain=%s",
            $url
        ));
    }

    function dum_row_is_up($row) {
        $code = (int) $row->status;
        return $code > 0 && $code < 600;
    }

    function dum_status_sort_label($row) {
        return dum_row_is_up($row) ? 'UP' : 'DOWN';
    }

    function dum_format_last_checked($last_checked) {
        $last_checked = trim((string) $last_checked);
        if ($last_checked === '') {
            return '';
        }
        try {
            $tz = new DateTimeZone('Asia/Manila');
            $dt = DateTime::createFromFormat('Y-m-d H:i:s', $last_checked, $tz);
            if ($dt instanceof DateTime) {
                $dt->setTimezone($tz);
                return $dt->format('F j, Y g:i a');
            }
        } catch (Exception $e) {
            // Fall through to strtotime fallback.
        }
        $ts = strtotime($last_checked);
        if ($ts === false) {
            return $last_checked;
        }
        $dt = new DateTime('@' . $ts);
        $dt->setTimezone(new DateTimeZone('Asia/Manila'));
        return $dt->format('F j, Y g:i a');
    }

    function dum_check_batch_size() {
        $n = (int) apply_filters('dum_check_batch_size', (int) DUM_CHECK_BATCH_SIZE);
        return max(3, min(80, $n));
    }

    function dum_check_request_time_budget() {
        $t = (float) apply_filters('dum_check_request_time_budget', (float) DUM_CHECK_TIME_BUDGET);
        return max(5.0, min(120.0, $t));
    }

    /**
     * Probe up to $limit rows with id > $after_id (stable cursor for batching).
     *
     * @return array{checked:int,last_id:int,done:bool}
     */
    function dum_process_probe_batch($wpdb, $table_name, $after_id, $limit) {
        $after_id = (int) $after_id;
        $limit = (int) $limit;
        $rows = $wpdb->get_results($wpdb->prepare(
            "SELECT id, domain FROM $table_name WHERE id > %d ORDER BY id ASC LIMIT %d",
            $after_id,
            $limit
        ));
        if (empty($rows)) {
            return ['checked' => 0, 'last_id' => $after_id, 'done' => true];
        }
        $checked = 0;
        $last_id = $after_id;
        foreach ($rows as $row) {
            $last_id = (int) $row->id;
            $url = dum_normalize_domain($row->domain);
            if (!$url) {
                continue;
            }
            dum_upsert_domain($wpdb, $table_name, $url);
            $checked++;
        }
        $done = count($rows) < $limit;
        return ['checked' => $checked, 'last_id' => $last_id, 'done' => $done];
    }

    /**
     * Run as many batches as fit in $budget_seconds (wall clock).
     *
     * @return array{checked:int,done:bool,resume_after_id:int}
     */
    function dum_run_probe_until_budget($wpdb, $table_name, $start_after_id, $budget_seconds) {
        $deadline = microtime(true) + max(2.0, (float) $budget_seconds);
        $after_id = (int) $start_after_id;
        $total_checked = 0;
        $batch_size = dum_check_batch_size();
        while (microtime(true) < $deadline) {
            $b = dum_process_probe_batch($wpdb, $table_name, $after_id, $batch_size);
            $total_checked += (int) $b['checked'];
            $after_id = (int) $b['last_id'];
            if (!empty($b['done'])) {
                return [
                    'checked' => $total_checked,
                    'done' => true,
                    'resume_after_id' => 0,
                ];
            }
        }
        return [
            'checked' => $total_checked,
            'done' => false,
            'resume_after_id' => $after_id,
        ];
    }

    /**
     * Full table probe (CLI / legacy POST). Loops time slices to avoid one huge gateway window.
     */
    function dum_run_check_all_domains($wpdb, $table_name) {
        @set_time_limit(0);
        if (function_exists('ignore_user_abort')) {
            ignore_user_abort(true);
        }
        $ini = (int) ini_get('max_execution_time');
        $slice = $ini > 0 ? max(20.0, (float) $ini - 12) : 90.0;
        $slice = min($slice, 120.0);
        $after = 0;
        $total = 0;
        while (true) {
            $r = dum_run_probe_until_budget($wpdb, $table_name, $after, $slice);
            $total += (int) $r['checked'];
            if (!empty($r['done'])) {
                break;
            }
            $after = (int) $r['resume_after_id'];
        }
        return $total;
    }

    function dum_auto_check_interval_seconds() {
        return 10 * MINUTE_IN_SECONDS;
    }

    function dum_ph_now_timestamp() {
        try {
            $tz = new DateTimeZone('Asia/Manila');
            $now = new DateTime('now', $tz);
            return (int) $now->getTimestamp();
        } catch (Exception $e) {
            return time();
        }
    }

    function dum_next_slot_timestamp($from_ts = null) {
        $interval = dum_auto_check_interval_seconds();
        $ts = $from_ts === null ? dum_ph_now_timestamp() : max(0, (int) $from_ts);
        return (int) ((floor($ts / $interval) + 1) * $interval);
    }

    function dum_auto_check_slot($ts) {
        $interval = dum_auto_check_interval_seconds();
        $ts = max(0, (int) $ts);
        return (int) floor($ts / $interval);
    }

    function dum_next_run_timestamp($wpdb, $table_name) {
        $now_ts = dum_ph_now_timestamp();
        $next_ts = dum_next_slot_timestamp($now_ts);
        update_option('dum_next_auto_run_ts', $next_ts, false);
        return $next_ts;
    }

    function dum_maybe_run_scheduled_check($wpdb, $table_name, $now_ts) {
        $now_ts = (int) $now_ts;
        if (get_transient('dum_scheduled_probe_state')) {
            return 0;
        }
        $current_slot = dum_auto_check_slot($now_ts);
        $last_checked_slot = (int) get_option('dum_last_auto_check_slot', -1);
        if ($last_checked_slot >= $current_slot) {
            return 0;
        }

        $lock_key = 'dum_auto_check_lock';
        if (get_transient($lock_key)) {
            return 0;
        }
        set_transient($lock_key, 1, 2 * HOUR_IN_SECONDS);

        try {
            $fresh_slot = dum_auto_check_slot(time());
            $fresh_last_slot = (int) get_option('dum_last_auto_check_slot', -1);
            if ($fresh_last_slot >= $fresh_slot) {
                return 0;
            }
            $budget = dum_check_request_time_budget();
            $r = dum_run_probe_until_budget($wpdb, $table_name, 0, $budget);
            if (!empty($r['done'])) {
                update_option('dum_last_auto_check_ts', $now_ts, false);
                update_option('dum_last_auto_check_slot', $fresh_slot, false);
                update_option('dum_next_auto_run_ts', dum_next_slot_timestamp($now_ts), false);
                return (int) $r['checked'];
            }
            set_transient('dum_scheduled_probe_state', [
                'active' => 1,
                'after_id' => (int) $r['resume_after_id'],
                'checked_total' => (int) $r['checked'],
                'target_slot' => $fresh_slot,
                'now_ts' => $now_ts,
            ], 6 * HOUR_IN_SECONDS);
            if (!wp_next_scheduled('dum_checker_scheduled_probe_resume')) {
                wp_schedule_single_event(time() + 60, 'dum_checker_scheduled_probe_resume');
            }
            return (int) $r['checked'];
        } finally {
            delete_transient($lock_key);
        }
    }

    function dum_continue_scheduled_probe_job($wpdb, $table_name) {
        $state = get_transient('dum_scheduled_probe_state');
        if (!is_array($state) || empty($state['active'])) {
            return 0;
        }
        $lock_key = 'dum_auto_check_lock';
        if (get_transient($lock_key)) {
            if (!wp_next_scheduled('dum_checker_scheduled_probe_resume')) {
                wp_schedule_single_event(time() + 90, 'dum_checker_scheduled_probe_resume');
            }
            return 0;
        }
        set_transient($lock_key, 1, 2 * HOUR_IN_SECONDS);
        try {
            $after = (int) ($state['after_id'] ?? 0);
            $r = dum_run_probe_until_budget($wpdb, $table_name, $after, dum_check_request_time_budget());
            $checked = (int) $r['checked'];
            $state['checked_total'] = (int) ($state['checked_total'] ?? 0) + $checked;
            $state['after_id'] = (int) $r['resume_after_id'];

            if (!empty($r['done'])) {
                delete_transient('dum_scheduled_probe_state');
                $now_ts = (int) ($state['now_ts'] ?? dum_ph_now_timestamp());
                $fresh_slot = (int) ($state['target_slot'] ?? dum_auto_check_slot($now_ts));
                update_option('dum_last_auto_check_ts', $now_ts, false);
                update_option('dum_last_auto_check_slot', $fresh_slot, false);
                update_option('dum_next_auto_run_ts', dum_next_slot_timestamp($now_ts), false);
                wp_clear_scheduled_hook('dum_checker_scheduled_probe_resume');
            } else {
                set_transient('dum_scheduled_probe_state', $state, 6 * HOUR_IN_SECONDS);
                if (!wp_next_scheduled('dum_checker_scheduled_probe_resume')) {
                    wp_schedule_single_event(time() + 60, 'dum_checker_scheduled_probe_resume');
                }
            }
            return $checked;
        } finally {
            delete_transient($lock_key);
        }
    }

    function dum_on_scheduled_probe_resume() {
        global $wpdb;
        dum_ensure_table();
        dum_continue_scheduled_probe_job($wpdb, dum_table_name());
    }

    function dum_maybe_run_target_slot_check($wpdb, $table_name, $now_ts, $target_slot) {
        $now_ts = (int) $now_ts;
        $target_slot = (int) $target_slot;
        if ($target_slot < 0) {
            return 0;
        }

        $last_checked_slot = (int) get_option('dum_last_auto_check_slot', -1);
        if ($last_checked_slot >= $target_slot) {
            return 0;
        }

        $lock_key = 'dum_auto_check_lock';
        if (get_transient($lock_key)) {
            return 0;
        }
        set_transient($lock_key, 1, dum_auto_check_interval_seconds());

        try {
            $fresh_last_slot = (int) get_option('dum_last_auto_check_slot', -1);
            if ($fresh_last_slot >= $target_slot) {
                return 0;
            }
            $checked = dum_run_check_all_domains($wpdb, $table_name);
            $effective_slot = max(dum_auto_check_slot($now_ts), $target_slot);
            update_option('dum_last_auto_check_ts', $now_ts, false);
            update_option('dum_last_auto_check_slot', $effective_slot, false);
            update_option('dum_next_auto_run_ts', dum_next_slot_timestamp($now_ts), false);
            return $checked;
        } finally {
            delete_transient($lock_key);
        }
    }

    function dum_schedule_auto_check_event() {
        $hook = 'dum_checker_auto_check_cron';
        $schedule = 'dum_every_ten_minutes';
        $next = wp_next_scheduled($hook);
        if ($next) {
            $current = wp_get_schedule($hook);
            if ($current && $current !== $schedule) {
                wp_unschedule_event($next, $hook);
                $next = false;
            }
        }
        if (!$next) {
            $first_run = dum_next_slot_timestamp();
            wp_schedule_event($first_run, $schedule, $hook);
        }
    }

    function dum_auto_check_cron_runner() {
        global $wpdb;
        dum_ensure_table();
        $table_name = dum_table_name();
        if (dum_is_sheet_source_mode()) {
            dum_sync_domains_from_google_sheet($wpdb, $table_name, true);
        }
        dum_continue_scheduled_probe_job($wpdb, $table_name);
        $now_ts = dum_ph_now_timestamp();
        dum_maybe_run_scheduled_check($wpdb, $table_name, $now_ts);
        update_option('dum_next_auto_run_ts', dum_next_slot_timestamp($now_ts), false);
    }

    function dum_ajax_verify_nonce() {
        if (empty($_POST['nonce']) || !wp_verify_nonce(sanitize_text_field(wp_unslash($_POST['nonce'])), 'domain_uptime_ajax')) {
            wp_send_json_error(['message' => 'Invalid security token.'], 403);
        }
    }

    function dum_ajax_add() {
        dum_ajax_verify_nonce();
        if (dum_is_sheet_source_mode()) {
            wp_send_json_error(['message' => 'Ang listahan ay mula sa Google Sheet; mag-edit doon.'], 403);
        }
        global $wpdb;
        dum_ensure_table();
        $table_name = dum_table_name();
        $raw = isset($_POST['domain_url']) ? wp_unslash($_POST['domain_url']) : '';
        $url = dum_normalize_domain($raw);
        if (!$url) {
            wp_send_json_error(['message' => 'Invalid domain.']);
        }
        $row = dum_upsert_domain($wpdb, $table_name, $url);
        if (!$row) {
            wp_send_json_error(['message' => 'Could not save domain.']);
        }
        wp_send_json_success([
            'row_html' => dum_render_data_row_html($row, dum_is_sheet_source_mode()),
            'row_id' => (int) $row->id,
            'count' => dum_count_rows(),
        ]);
    }

    function dum_ajax_delete() {
        dum_ajax_verify_nonce();
        if (dum_is_sheet_source_mode()) {
            wp_send_json_error(['message' => 'Ang listahan ay mula sa Google Sheet; mag-edit doon.'], 403);
        }
        global $wpdb;
        $id = isset($_POST['id']) ? absint($_POST['id']) : 0;
        if (!$id) {
            wp_send_json_error(['message' => 'Invalid row.']);
        }
        dum_ensure_table();
        $wpdb->delete(dum_table_name(), ['id' => $id], ['%d']);
        wp_send_json_success(['count' => dum_count_rows()]);
    }

    function dum_ajax_delete_selected() {
        dum_ajax_verify_nonce();
        if (dum_is_sheet_source_mode()) {
            wp_send_json_error(['message' => 'Ang listahan ay mula sa Google Sheet; mag-edit doon.'], 403);
        }
        global $wpdb;
        $ids = isset($_POST['ids']) ? (array) $_POST['ids'] : [];
        $ids = array_values(array_filter(array_map('absint', $ids)));
        if (empty($ids)) {
            wp_send_json_error(['message' => 'No rows selected.']);
        }
        dum_ensure_table();
        $table = dum_table_name();

        $placeholders = implode(',', array_fill(0, count($ids), '%d'));
        $sql = "DELETE FROM $table WHERE id IN ($placeholders)";
        $wpdb->query($wpdb->prepare($sql, $ids));

        wp_send_json_success(['count' => dum_count_rows()]);
    }

    function dum_ajax_auto_check() {
        dum_ajax_verify_nonce();
        global $wpdb;
        dum_ensure_table();
        $table_name = dum_table_name();
        $vo = dum_is_sheet_source_mode();
        if ($vo) {
            dum_sync_domains_from_google_sheet($wpdb, $table_name, false);
        }
        $server_now = dum_ph_now_timestamp();
        $target_slot = isset($_POST['target_slot']) ? (int) wp_unslash($_POST['target_slot']) : -1;
        $manual_full = isset($_POST['manual_full']) && (string) wp_unslash($_POST['manual_full']) === '1';
        $budget = dum_check_request_time_budget();
        $checked = 0;
        $probe_done = true;
        $probe_after_id = 0;
        $sheet_probe_continue = false;

        if ($manual_full) {
            $after = isset($_POST['probe_after_id']) ? (int) wp_unslash($_POST['probe_after_id']) : 0;
            $r = dum_run_probe_until_budget($wpdb, $table_name, $after, $budget);
            $checked = (int) $r['checked'];
            $probe_done = !empty($r['done']);
            $probe_after_id = $probe_done ? 0 : (int) $r['resume_after_id'];
            if ($probe_done) {
                update_option('dum_last_auto_check_ts', $server_now, false);
                update_option('dum_last_auto_check_slot', dum_auto_check_slot($server_now), false);
                update_option('dum_next_auto_run_ts', dum_next_slot_timestamp($server_now), false);
            }
        } elseif ($target_slot >= 0) {
            $after = isset($_POST['probe_after_id']) ? (int) wp_unslash($_POST['probe_after_id']) : 0;
            $r = dum_run_probe_until_budget($wpdb, $table_name, $after, $budget);
            $checked = (int) $r['checked'];
            $probe_done = !empty($r['done']);
            $probe_after_id = $probe_done ? 0 : (int) $r['resume_after_id'];
            if ($probe_done) {
                $slot = dum_auto_check_slot($server_now);
                update_option('dum_last_auto_check_ts', $server_now, false);
                update_option('dum_last_auto_check_slot', $slot, false);
                update_option('dum_next_auto_run_ts', dum_next_slot_timestamp($server_now), false);
            }
        } else {
            $checked = dum_maybe_run_scheduled_check($wpdb, $table_name, $server_now);
            if ($vo && $checked === 0) {
                $iv = dum_auto_check_interval_seconds();
                $sheet_after = isset($_POST['sheet_probe_after_id']) ? (int) wp_unslash($_POST['sheet_probe_after_id']) : 0;
                $due = ($server_now - (int) get_option('dum_sheet_ui_probe_ts', 0) >= $iv) || $sheet_after > 0;
                if ($due) {
                    $r = dum_run_probe_until_budget($wpdb, $table_name, $sheet_after, $budget);
                    $checked = (int) $r['checked'];
                    $probe_done = !empty($r['done']);
                    $probe_after_id = $probe_done ? 0 : (int) $r['resume_after_id'];
                    $sheet_probe_continue = !$probe_done;
                    if ($probe_done) {
                        update_option('dum_sheet_ui_probe_ts', $server_now, false);
                    }
                }
            }
        }

        $rows = $wpdb->get_results("SELECT * FROM $table_name ORDER BY last_checked DESC");

        $omit_rows_html = false;
        if ($manual_full || $target_slot >= 0) {
            $omit_rows_html = !$probe_done;
        } elseif ($vo && $sheet_probe_continue) {
            $omit_rows_html = true;
        }

        wp_send_json_success([
            'checked' => (int) $checked,
            'count' => dum_count_rows(),
            'next_run_ts' => (int) dum_next_run_timestamp($wpdb, $table_name),
            'rows_omitted' => $omit_rows_html,
            'rows_html' => $omit_rows_html ? '' : dum_render_table_rows_html($rows, $vo),
            'probe_done' => $probe_done,
            'probe_after_id' => $probe_after_id,
            'sheet_probe_continue' => $sheet_probe_continue,
        ]);
    }

    add_action('wp_ajax_dum_checker_add', 'dum_ajax_add');
    add_action('wp_ajax_dum_checker_delete', 'dum_ajax_delete');
    add_action('wp_ajax_dum_checker_delete_selected', 'dum_ajax_delete_selected');
    add_action('wp_ajax_dum_checker_auto_check', 'dum_ajax_auto_check');
    add_action('wp_ajax_nopriv_dum_checker_auto_check', 'dum_ajax_auto_check');
    add_filter('cron_schedules', function ($schedules) {
        $schedules['dum_every_ten_minutes'] = [
            'interval' => dum_auto_check_interval_seconds(),
            'display' => 'Every 10 Minutes (DUM)',
        ];
        return $schedules;
    });
    add_action('init', 'dum_schedule_auto_check_event');
    add_action('dum_checker_auto_check_cron', 'dum_auto_check_cron_runner');
    add_action('dum_checker_scheduled_probe_resume', 'dum_on_scheduled_probe_resume');
}

    if (autoCountdown) {

        if (!isNaN(_iv) && _iv > 0) {
            autoIntervalSec = _iv;

        var serverNowTs = parseInt(autoCountdown.getAttribute('data-server-now') || '0', 10);
        if (!isNaN(serverNowTs) && serverNowTs > 0) {
            countdownSkewMs = Date.now() - (serverNowTs * 1000);

    function monitorClientNowTs() {
        return Math.floor((Date.now() - countdownSkewMs) / 1000);
    }

    function monitorFormatCountdown(totalSec) {
        var sec = Math.max(0, parseInt(totalSec, 10) || 0);
        if (sec >= 3600) {
            var h = Math.floor(sec / 3600);
            var m = Math.floor((sec % 3600) / 60);
            var s = sec % 60;
            return h + ':' + String(m).padStart(2, '0') + ':' + String(s).padStart(2, '0');
        }
        var mm = Math.floor(sec / 60);
        var ss = sec % 60;
        return String(mm).padStart(2, '0') + ':' + String(ss).padStart(2, '0');
    }

    function monitorApplyAutoCheckPayload(data) {
        if (typeof data.count !== 'undefined') {
            monitorUpdateStatCount(parseInt(data.count, 10) || 0);
        }
        if (autoCountdown) {
            if (typeof data.next_run_ts !== 'undefined') {
                var nextTs = parseInt(data.next_run_ts, 10) || 0;
                if (nextTs > monitorClientNowTs()) {
                    autoCountdown.setAttribute('data-next-run-ts', String(nextTs));
                } else {
                    autoCountdown.setAttribute('data-next-run-ts', String(monitorClientNowTs() + autoIntervalSec));
                }
            } else {
                autoCountdown.setAttribute('data-next-run-ts', String(monitorClientNowTs() + autoIntervalSec));
            }
        }
        if (data.rows_omitted !== true && typeof data.rows_html === 'string') {
            var tbody = table.querySelector('tbody');
            if (tbody) {
                if (data.rows_html.trim() === '') {
                    tbody.innerHTML = '';
                    monitorEnsureEmptyPlaceholder();
                } else {
                    tbody.innerHTML = data.rows_html;
                }
                currentPage = 1;
                monitorRebuildView();
                monitorUpdateSelectedUI();
            }
        }
    }

    function monitorRunProbeChunks(isManual, targetSlot, startAfter) {
        if (!autoCountdown || !ajaxUrl || !ajaxNonce || autoSubmitLock) return;
        autoSubmitLock = true;
        function finishOk() {
            autoSubmitLock = false;
            monitorUpdateAutoCountdown();
            if (monitorCheckAllBtn) {
                monitorCheckAllBtn.disabled = false;
                var lbl = monitorCheckAllBtn.getAttribute('data-dum-check-label');
                if (lbl) monitorCheckAllBtn.textContent = lbl;
            }
        }
        function failFallback() {
            if (autoCheckForm) {
                if (isManual) {
                    if (monitorCheckAllBtn) {
                        monitorCheckAllBtn.disabled = false;
                        var lbl0 = monitorCheckAllBtn.getAttribute('data-dum-check-label');
                        if (lbl0) monitorCheckAllBtn.textContent = lbl0;
                    }
                    autoCheckForm.submit();
                    return;
                }
                var autoTickInput2 = autoCheckForm.querySelector('input[name="dum_auto_tick"]');
                if (!autoTickInput2) {
                    autoTickInput2 = document.createElement('input');
                    autoTickInput2.type = 'hidden';
                    autoTickInput2.name = 'dum_auto_tick';
                    autoCheckForm.appendChild(autoTickInput2);
                }
                autoTickInput2.value = '1';
                autoCheckForm.submit();
                return;
            }
            if (autoCountdown) {
                autoCountdown.setAttribute('data-next-run-ts', String(monitorClientNowTs() + autoIntervalSec));
            }
            autoSubmitLock = false;
            monitorUpdateAutoCountdown();
        }
        function step(aid) {
            var body = new URLSearchParams();
            body.set('action', 'dum_checker_auto_check');
            body.set('nonce', ajaxNonce);
            body.set('probe_after_id', String(aid));
            if (isManual) {
                body.set('manual_full', '1');
            } else {
                body.set('target_slot', String(targetSlot));
            }
            fetch(ajaxUrl, {
                method: 'POST',
                credentials: 'same-origin',
                headers: { 'Content-Type': 'application/x-www-form-urlencoded; charset=UTF-8' },
                body: body.toString()
            })
                .then(function (r) { return r.json(); })
                .then(function (res) {
                    if (!res || !res.success || !res.data) {
                        failFallback();
                        return;
                    }
                    var data = res.data;
                    monitorApplyAutoCheckPayload(data);
                    var done = data.probe_done !== false && data.probe_done !== '0';
                    if (!done) {
                        step(parseInt(data.probe_after_id, 10) || 0);
                        return;
                    }
                    finishOk();
                })
                .catch(function () {
                    failFallback();
                });
        }
        step(startAfter || 0);
    }

    function monitorRequestAutoCheck(targetSlot) {

        if (typeof targetSlot === 'number' && targetSlot >= 0) {
            monitorRunProbeChunks(false, targetSlot, 0);
            return;
        }

        body.set('action', 'dum_checker_auto_check');
        body.set('nonce', ajaxNonce);

            body.set('sheet_probe_after_id', String(sheetProbeAfterId));
        }

        fetch(ajaxUrl, {
            method: 'POST',
            credentials: 'same-origin',
            headers: { 'Content-Type': 'application/x-www-form-urlencoded; charset=UTF-8' },
            body: body.toString()
        })
            .then(function (r) { return r.json(); })
            .then(function (res) {
                if (res && res.success && res.data) {
                    var data = res.data;
                    monitorApplyAutoCheckPayload(data);
                    if (data.sheet_probe_continue) {
                        sheetProbeAfterId = parseInt(data.probe_after_id, 10) || 0;
                        autoSubmitLock = false;
                        monitorUpdateAutoCountdown();
                        monitorRequestAutoCheck(-1);
                        return;
                    }
                    sheetProbeAfterId = 0;
                } else {
                    if (autoCheckForm) {
                        var autoTickInput2 = autoCheckForm.querySelector('input[name="dum_auto_tick"]');
                        if (!autoTickInput2) {
                            autoTickInput2 = document.createElement('input');
                            autoTickInput2.type = 'hidden';
                            autoTickInput2.name = 'dum_auto_tick';
                            autoCheckForm.appendChild(autoTickInput2);
                        }
                        autoTickInput2.value = '1';
                        autoCheckForm.submit();
                        return;
                    }
                    autoCountdown.setAttribute('data-next-run-ts', String(monitorClientNowTs() + autoIntervalSec));
                }
                autoSubmitLock = false;
                monitorUpdateAutoCountdown();
            })
            .catch(function () {
                if (autoCheckForm) {
                    var autoTickInput3 = autoCheckForm.querySelector('input[name="dum_auto_tick"]');
                    if (!autoTickInput3) {
                        autoTickInput3 = document.createElement('input');
                        autoTickInput3.type = 'hidden';
                        autoTickInput3.name = 'dum_auto_tick';
                        autoCheckForm.appendChild(autoTickInput3);
                    }
                    autoTickInput3.value = '1';
                    autoCheckForm.submit();
                    return;
                }
                autoCountdown.setAttribute('data-next-run-ts', String(monitorClientNowTs() + autoIntervalSec));
                autoSubmitLock = false;
            });
    }

    if (autoCheckForm) {

        });

    function monitorUpdateAutoCountdown() {
        if (!autoCountdown) return;
        var nextRunTs = parseInt(autoCountdown.getAttribute('data-next-run-ts') || '0', 10);
        if (isNaN(nextRunTs) || nextRunTs <= 0) return;
        var nowTs = monitorClientNowTs();
        var remaining = nextRunTs - nowTs;
        autoCountdown.textContent = monitorFormatCountdown(remaining);
        if (remaining <= 0) {
            autoCountdown.textContent = '00:00';
            // Move target forward immediately to avoid repeat triggers while request is in flight.
            autoCountdown.setAttribute('data-next-run-ts', String(nowTs + autoIntervalSec));
            var targetSlot = Math.floor(nextRunTs / autoIntervalSec);
            monitorRequestAutoCheck(targetSlot);
        }
    }

    if (autoCountdown) {
        monitorUpdateAutoCountdown();
        autoCountdownTimer = window.setInterval(monitorUpdateAutoCountdown, 1000);
        window.setInterval(function () {
            if (document.visibilityState === 'hidden') return;
            monitorRequestAutoCheck(-1);
        }, 15000);
    }

        table.querySelectorAll('tbody tr.checker-data-row input.checker-row-select:checked').forEach(function (cb) {

                if (cb) visibleCbs.push(cb);

                selectAll.checked = false;
                selectAll.indeterminate = false;
            } else {
                var checkedCount = visibleCbs.filter(function (cb) { return cb.checked; }).length;
                selectAll.checked = checkedCount === visibleCbs.length;
                selectAll.indeterminate = checkedCount > 0 && checkedCount < visibleCbs.length;
            }

    function monitorRunPendingDelete() {
        if (!pendingDelete || !ajaxUrl || !ajaxNonce) return;
        var kind = pendingDelete.kind || 'single';
        var link = pendingDelete.link || null;
        var id = pendingDelete.id || null;
        var ids = pendingDelete.ids || null;
        monitorCloseDeleteModal();
        var body = new URLSearchParams();
        body.set('action', kind === 'bulk' ? 'dum_checker_delete_selected' : 'dum_checker_delete');
        body.set('nonce', ajaxNonce);
        if (kind === 'bulk') {
            (ids || []).forEach(function (x) { body.append('ids[]', x); });
        } else {
            body.set('id', id);
        }
        if (link) link.style.pointerEvents = 'none';
        fetch(ajaxUrl, {
            method: 'POST',
            credentials: 'same-origin',
            headers: { 'Content-Type': 'application/x-www-form-urlencoded; charset=UTF-8' },
            body: body.toString()
        })
            .then(function (r) { return r.json(); })
            .then(function (res) {
                if (link) link.style.pointerEvents = '';
                if (!res || !res.success) {
                    var msg = (res && res.data && res.data.message) ? res.data.message : 'Could not delete.';
                    window.alert(msg);
                    return;
                }
                if (kind === 'bulk') {
                    (ids || []).forEach(function (x) {
                        var trb = table.querySelector('tbody tr.checker-data-row[data-row-id=\"' + String(x) + '\"]');
                        if (trb) trb.remove();
                    });
                } else if (link) {
                    var tr = link.closest('tr.checker-data-row');
                    if (tr) tr.remove();
                }
                monitorEnsureEmptyPlaceholder();
                monitorUpdateStatCount(res.data.count);
                monitorRebuildView();
                monitorUpdateSelectedUI();
            })
            .catch(function () {
                link.style.pointerEvents = '';
                window.alert('Network error.');
            });
    }

    if (deleteModalCancel) {
        deleteModalCancel.addEventListener('click', monitorCloseDeleteModal);
    }
    if (deleteModalBackdrop) {
        deleteModalBackdrop.addEventListener('click', monitorCloseDeleteModal);
    }
    if (deleteModalConfirm) {
        deleteModalConfirm.addEventListener('click', function () {
            if (!pendingDelete) return;
            if (deleteModalCancel) deleteModalCancel.disabled = true;
            deleteModalConfirm.disabled = true;
            monitorRunPendingDelete();
        });
    }

    if (ajaxUrl && ajaxNonce) {
        table.addEventListener('click', function (e) {
            var link = e.target.closest('a.checker-delete');
            if (!link || !table.contains(link)) return;
            e.preventDefault();
            var id = link.getAttribute('data-delete-id');
            if (!id) return;
            var delRow = link.closest('tr.checker-data-row');
            var domainCell = delRow ? delRow.querySelector('.checker-cell-domain') : null;
            var domainLabel = domainCell ? domainCell.textContent.trim() : '';
            if (deleteModal && deleteModalDomain) {
                monitorOpenDeleteModal(link, id, domainLabel);
            } else {
                if (!window.confirm(domainLabel ? 'Remove ' + domainLabel + ' from the list?' : 'Remove this site from the list?')) return;
                pendingDelete = { link: link, id: id };
                monitorRunPendingDelete();
            }
        });
    }

    table.addEventListener('change', function (e) {
        if (e.target && e.target.classList && e.target.classList.contains('checker-row-select')) {
            monitorUpdateSelectedUI();
        }
    });

    if (selectAll) {
        selectAll.addEventListener('change', function () {
            var want = !!selectAll.checked;
            table.querySelectorAll('tbody tr.checker-data-row').forEach(function (row) {
                if (row.style.display === 'none') return;
                var cb = row.querySelector('input.checker-row-select');
                if (cb) cb.checked = want;
            });
            monitorUpdateSelectedUI();
        });
    }

    if (deleteSelectedBtn && ajaxUrl && ajaxNonce) {
        deleteSelectedBtn.addEventListener('click', function () {
            var ids = monitorSelectedIds();
            if (!ids.length) return;
            if (deleteModal && deleteModalDomain) {
                pendingDelete = { kind: 'bulk', ids: ids.slice() };
                monitorConfigureDeleteModal(
                    ids.length === 1 ? 'Remove selected site' : 'Remove selected sites',
                    ids.length === 1
                        ? 'Remove the selected site from your monitoring list?'
                        : ('Remove ' + ids.length + ' selected sites from your monitoring list?'),
                    ids.length === 1 ? '1 site selected' : (ids.length + ' sites selected'),
                    ids.length === 1 ? 'Remove' : 'Remove all'
                );
                deleteModal.removeAttribute('hidden');
                deleteModal.setAttribute('aria-hidden', 'false');
                document.addEventListener('keydown', monitorDeleteModalKeydown);
                if (deleteModalCancel) deleteModalCancel.focus();
            } else {
                // fallback
                if (!window.confirm(ids.length === 1 ? 'Delete the selected site?' : ('Delete ' + ids.length + ' selected sites?'))) return;
                pendingDelete = { kind: 'bulk', ids: ids.slice() };
                monitorRunPendingDelete();
            }
        });
    }

    function monitorApplyStripes() {
        var tbody = table.querySelector('tbody');
        if (!tbody) return;
        var n = 0;
        tbody.querySelectorAll('tr.checker-data-row').forEach(function (row) {
            row.classList.remove('checker-row-light', 'checker-row-dark');
            if (row.style.display === 'none') return;
            n += 1;
            row.classList.add(n % 2 === 1 ? 'checker-row-light' : 'checker-row-dark');
        });
    }

    function monitorRebuildView() {
        var tbody = table.querySelector('tbody');
        if (!tbody) return;

        var allRows = Array.prototype.slice.call(tbody.querySelectorAll('tr.checker-data-row'));
        var filterActive = search && search.value.trim() !== '';

        if (filterActive) {
            if (pagBar) pagBar.style.display = 'none';
            var ft = search.value.toLowerCase().trim();
            allRows.forEach(function (row) {
                row.style.display = row.innerText.toLowerCase().includes(ft) ? '' : 'none';
            });
        } else {
            var total = allRows.length;
            var perPageValue = perPageSel ? perPageSel.value : '10';
            var perPage = perPageValue === 'all' ? total : (parseInt(perPageValue, 10) || 10);

            if (pagBar) {
                pagBar.style.display = total === 0 ? 'none' : 'grid';
            }

            if (total > 0) {
                var totalPages = perPageValue === 'all' ? 1 : Math.max(1, Math.ceil(total / perPage));
                if (currentPage > totalPages) currentPage = totalPages;
                if (currentPage < 1) currentPage = 1;

                allRows.forEach(function (row, i) {
                    var onPage = i >= (currentPage - 1) * perPage && i < currentPage * perPage;
                    row.style.display = onPage ? '' : 'none';
                });

                if (pagBar && pageInfo) {
                    var start = perPageValue === 'all' ? 1 : ((currentPage - 1) * perPage + 1);
                    var end = perPageValue === 'all' ? total : Math.min(currentPage * perPage, total);
                    pageInfo.textContent = 'Showing ' + start + '–' + end + ' of ' + total + ' · Page ' + currentPage + ' / ' + totalPages;
                }
                if (pagePrev) pagePrev.disabled = currentPage <= 1 || perPageValue === 'all';
                if (pageNext) pageNext.disabled = currentPage >= totalPages || perPageValue === 'all';
            }
        }

        monitorApplyStripes();
        monitorUpdateSelectedUI();
    }

    function monitorSortValue(row, col) {
        var map = {
            domain: 'data-sort-domain',
            status: 'data-sort-status-label',
            detail: 'data-sort-detail',
            time: 'data-sort-time'
        };
        var attr = map[col];
        if (!attr) return '';
        var v = row.getAttribute(attr);
        return v === null ? '' : v;
    }

    function monitorSortTable(col, dir) {
        var tbody = table.querySelector('tbody');
        if (!tbody) return;
        var rows = Array.prototype.slice.call(tbody.querySelectorAll('tr.checker-data-row'));
        if (rows.length === 0) return;

        var asc = dir === 'asc';

        rows.sort(function (a, b) {
            // Custom ordering for Status: for DESC we want DOWN first (worst-first).
            if (col === 'status') {
                var ra = String(monitorSortValue(a, col) || '').toUpperCase();
                var rb = String(monitorSortValue(b, col) || '').toUpperCase();
                var rank = { 'DOWN': 0, 'UP': 1 };
                var na = Object.prototype.hasOwnProperty.call(rank, ra) ? rank[ra] : 2;
                var nb = Object.prototype.hasOwnProperty.call(rank, rb) ? rank[rb] : 2;
                // asc: UP first; desc: DOWN first
                var cmpStatus = asc ? (nb - na) : (na - nb);
                if (cmpStatus !== 0) return cmpStatus;
                // stable-ish tie-breaker
                var da = monitorSortValue(a, 'domain');
                var db = monitorSortValue(b, 'domain');
                return String(da).localeCompare(String(db), undefined, { sensitivity: 'base', numeric: true });
            }

            var va = monitorSortValue(a, col);
            var vb = monitorSortValue(b, col);
            var cmp = String(va).localeCompare(String(vb), undefined, { sensitivity: 'base', numeric: true });
            return asc ? cmp : -cmp;
        });

        var emptyRow = tbody.querySelector('tr.checker-table-empty');
        rows.forEach(function (row) {
            tbody.appendChild(row);
        });
        if (emptyRow) {
            tbody.appendChild(emptyRow);
        }

        currentPage = 1;
        monitorRebuildView();
    }

    table.querySelectorAll('.checker-sort-btn').forEach(function (btn) {
        btn.addEventListener('click', function () {
            var col = btn.getAttribute('data-monitor-sort');
            var dir = btn.getAttribute('data-monitor-dir');
            if (col && dir) {
                monitorSortTable(col, dir);
            }
        });
    });

    if (search) {
        search.addEventListener('input', monitorRebuildView);
    }
    if (perPageSel) {
        perPageSel.addEventListener('change', function () {
            currentPage = 1;
            monitorRebuildView();
        });
    }
    if (pagePrev) {
        pagePrev.addEventListener('click', function () {
            if (currentPage > 1) {
                currentPage -= 1;
                monitorRebuildView();
            }
        });
    }
    if (pageNext) {
        pageNext.addEventListener('click', function () {
            var tbody = table.querySelector('tbody');
            if (!tbody) return;
            var total = tbody.querySelectorAll('tr.checker-data-row').length;
            var perPageValue = perPageSel ? perPageSel.value : '10';
            var perPage = perPageValue === 'all' ? total : (parseInt(perPageValue, 10) || 10);
            var totalPages = perPageValue === 'all' ? 1 : Math.max(1, Math.ceil(total / perPage));
            if (currentPage < totalPages) {
                currentPage += 1;
                monitorRebuildView();
            }
        });
    }

    // Initial priority: show DOWN sites first by default.
    monitorSortTable('status', 'desc');
})();
</script>
